# === SERVER COMMANDS - PROD ===

# Start the backend server (PROD)
function run-backend {
    expenses-backend
    Write-Host "[START-PROD] Starting backend server on port 8000..." -ForegroundColor Green
    $env:PORT = "8000"
    $env:ENVIRONMENT = "prod"
    $env:DATABASE_URL = "postgresql://postgres:expenses@localhost:5432/expenses_db"
    $env:AUTH_SERVICE_URL = "http://localhost:8001"
    # Activate venv if it exists
    if (Test-Path ".\venv\Scripts\Activate.ps1") {
        .\venv\Scripts\Activate.ps1
    }
    python dev_server.py
}

# Start the frontend dev server (PROD)
function run-frontend {
    expenses-frontend
    Write-Host "[START-PROD] Starting frontend dev server on port 5173..." -ForegroundColor Green
    $env:VITE_PORT = "5173"
    $env:VITE_API_URL = "http://localhost:8000"
    npm run dev
}

# Start the auth service (PROD)
function run-auth {
    expenses-auth
    Write-Host "[START-PROD] Starting auth service on port 8001..." -ForegroundColor Green
    $env:PORT = "8001"
    $env:DB_PORT = "5432"
    $env:DB_NAME = "expenses_db"
    $env:CORS_ORIGINS = "http://localhost:5173,http://localhost:3000,http://localhost:8000"
    # Activate venv if it exists
    if (Test-Path ".\venv\Scripts\Activate.ps1") {
        .\venv\Scripts\Activate.ps1
    }
    python dev_server.py
}

# Start all services in Windows Terminal tabs (PROD)
function run-all-tabs-prod {
    Write-Host "[START-PROD] Starting all PROD services in new tabs..." -ForegroundColor Cyan
    Write-Host "  Auth:     localhost:8001" -ForegroundColor Cyan
    Write-Host "  Backend:  localhost:8000" -ForegroundColor Cyan
    Write-Host "  Frontend: localhost:5173" -ForegroundColor Cyan
    Write-Host "  Database: localhost:5432 (expenses_db)" -ForegroundColor Cyan
    wt -w 0 new-tab --title "Auth-PROD"     powershell.exe -NoExit -Command "run-auth"
    wt -w 0 new-tab --title "Backend-PROD"  powershell.exe -NoExit -Command "run-backend"
    wt -w 0 new-tab --title "Frontend-PROD" powershell.exe -NoExit -Command "run-frontend"
    Write-Host "[SUCCESS] PROD services started!" -ForegroundColor Green
}

# Reverte até uma migration específica (PROD)
function db-rollback-to {
    param([string]$migration)
    if (-not $migration) {

    Write-Host "[DB-PROD] Revertendo até migration: $migration" -ForegroundColor Yellow
    $env:DATABASE_URL = "postgresql://postgres:expenses@localhost:5432/expenses_db"
    python .\services\backend\scripts\migration_runner.py --reverter-ate $migration
}

# === DEVELOPMENT UTILITIES ===

# Kill all project processes
function kill-all {
    Write-Host "[STOP] Stopping all Plumo processes..." -ForegroundColor Red
    Get-Process | Where-Object {$_.ProcessName -eq "python" -and $_.CommandLine -like "*dev_server*"} | Stop-Process -Force -ErrorAction SilentlyContinue
    Get-Process | Where-Object {$_.ProcessName -eq "node" -and $_.CommandLine -like "*vite*"} | Stop-Process -Force -ErrorAction SilentlyContinue
    Write-Host "[SUCCESS] All processes stopped" -ForegroundColor Green
}

# Check what's running on common ports
function check-ports {
    Write-Host "[CHECK] Checking common Plumo ports..." -ForegroundColor Cyan
    $ports = @(5173, 8000, 8001, 5432, 5433)
    foreach ($port in $ports) {
        $connection = Get-NetTCPConnection -LocalPort $port -ErrorAction SilentlyContinue
        if ($connection) {
            Write-Host "[OK] Port $port is in use" -ForegroundColor Green
        } else {
            Write-Host "[FREE] Port $port is free" -ForegroundColor Red
        }
    }
}

# Full development setup
function dev-setup {
    Write-Host "[SETUP] Setting up development environment..." -ForegroundColor Yellow

    # Check if all required tools are available
    $tools = @("python", "node", "npm", "git", "docker")
    foreach ($tool in $tools) {
        if (Get-Command $tool -ErrorAction SilentlyContinue) {
            Write-Host "[OK] $tool is available" -ForegroundColor Green
        } else {
            Write-Host "[ERROR] $tool is not available" -ForegroundColor Red
        }
    }

    # Install dependencies
    Write-Host "[INSTALL] Installing Python dependencies..." -ForegroundColor Cyan
    expenses
    python scripts/install_requirements.py

    Write-Host "[INSTALL] Installing frontend dependencies..." -ForegroundColor Cyan
    expenses-frontend
    npm install

    Write-Host "[SUCCESS] Development setup complete!" -ForegroundColor Green
}

# Quick health check
function expenses-health {
    Write-Host "[HEALTH] Plumo Platform System Check..." -ForegroundColor Cyan
    check-ports
    Write-Host ""
    git-status-all
    Write-Host ""
    Write-Host "[STATUS] Database Status:" -ForegroundColor Yellow
    Write-Host "PROD (5432): " -NoNewline
    if (Test-NetConnection localhost -Port 5432 -WarningAction SilentlyContinue -InformationLevel Quiet) {
        Write-Host "[OK] Running" -ForegroundColor Green
    } else {
        Write-Host "[ERROR] Not Running" -ForegroundColor Red
    }
    Write-Host "DEV  (5433): " -NoNewline
    if (Test-NetConnection localhost -Port 5433 -WarningAction SilentlyContinue -InformationLevel Quiet) {
        Write-Host "[OK] Running" -ForegroundColor Green
    } else {
        Write-Host "[ERROR] Not Running" -ForegroundColor Red
    }
}
